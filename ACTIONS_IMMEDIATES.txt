━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
 ACTION 2 : CONFIGURER L'ENVIRONNEMENT (15 minutes)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

   ⚙️ Commandes à exécuter :
   
   cd c:\xampp\htdocs\csar
   cp .env.example .env
   notepad .env
   
   À configurer dans .env :
   ✅ DB_DATABASE=csar_production
   ✅ DB_USERNAME=votre_user
   ✅ DB_PASSWORD=votre_password
   ✅ SESSION_DRIVER=database   (sessions stockées en BDD)
   ✅ CACHE_STORE=database     (cache stocké en BDD)
   ✅ MAIL_HOST=smtp.gmail.com (ou autre)
   ✅ MAIL_USERNAME=votre_email
   ✅ MAIL_PASSWORD=votre_password
   
   Puis :
   php artisan key:generate
   php artisan migrate          (crée les tables sessions + cache si besoin)


━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
 ACTION 3 : CONFIGURER LES BACKUPS (10 minutes)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

   💾 Sur Windows :
   
   Double-cliquer sur : scripts\backup\setup_backup.bat
   
   Ce script va :
   ✅ Vérifier les prérequis (PHP, MySQL)
   ✅ Tester la sauvegarde
   ✅ Créer une tâche planifiée (quotidienne 2h)
   
   Sur Linux :
   chmod +x scripts/backup/database_backup.php
   crontab -e
   Ajouter : 0 2 * * * php /chemin/vers/database_backup.php


━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
 ACTION 4 : LANCER LES TESTS (5 minutes)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

   🧪 Vérifier que tout fonctionne :
   
   php artisan test
   
   Résultat attendu :
   ✓ PASS  Tests\Feature\AuthenticationTest (12 tests)
   ✓ PASS  Tests\Feature\StockManagementTest (10 tests)
   
   Tests: 22 passed
   
   ✅ Si tous les tests passent → Tout est OK !
   ❌ Si des tests échouent → Vérifier la configuration


━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
 ACTION 5 : ACTIVER LE MONITORING (5 minutes)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

   📊 Surveillance système :
   
   php artisan system:monitor
   
   Résultat attendu :
   ✅ Database: OK (< 1000ms)
   ✅ Disk: OK (< 90%)
   ✅ Memory: OK (< 85%)
   ✅ Services: OK
   
   Pour automatiser (Windows - Planificateur de tâches) :
   Tâche : php artisan system:monitor
   Fréquence : Toutes les 5 minutes
